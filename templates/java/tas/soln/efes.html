Learning Goal: for loops and decomposition
Milestone: build a single column

Write code on the board. Start with pseudocode, turn to real code. Encourage students to work in pairs to program out the milestones. The hardest part is the fence-post bug when building a col, and when putting beepers in a column.

Solution

public class BuildCharlesBridge extends SuperKarel {

    public void run() {
        for (int i = 0; i < 3; i++) {
            buildColumn();
            moveToNextColumn();
        }
        // need to build the last column without moving further east
        buildColumn();
    }

    /* Method: buildColumn
     * -------------------
     * Fills a column (of height 5) with beepers.
     * Precondition: Karel is facing east at the bottom of an empty column.
     * Postcondition: Karel is facing east at the bottom of the column, which
     * is now filled with beepers. 
     */
    private void buildColumn() {
        turnLeft();
        placeBeepersUp();
        comeBackDown();
        turnLeft();
    }

    /* Method: placeBeepersUp
     * ----------------------
     * Puts 5 beepers vertically in a column (of height 5).
     * Precondition: Karel is facing north at the bottom of the column.
     * Postcondition: Karel is facing north at the top of the column, which
     * is now filled with beepers.
     */
    private void placeBeepersUp() {
        for (int i = 0; i < 4; i++) {
            putBeeper();
            move();
        }
        // need to put the last beeper without moving further north
        putBeeper();
    }

    /* Method: comeBackDown
     * --------------------
     * Returns Karel to the bottom of a column of height 5.
     * Precondition: Karel is facing north at the top of the column.
     * Postcondition: Karel is facing south at the bottom of the column.
     */
    private void comeBackDown() {
        turnLeft();
        turnLeft();
        for (int i = 0; i < 4; i++) {
            move();
        }
    }

    /* Method: moveToNextColumn
     * ------------------------
     * Moves Karel from the bottom of one column to the bottom of the next one,
     * which is 4 spaces to the east.
     * Precondition: Karel is facing east at the bottom of one column.
     * Postcondition: Karel is facing east at the bottom of the next column.
     */
    private void moveToNextColumn() {
        for (int i = 0; i < 4; i++) {
            move();
        }
    }

}
